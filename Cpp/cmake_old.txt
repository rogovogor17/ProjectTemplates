cmake_minimum_required(VERSION 3.16)
project(MyProject LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set (DEBUG_FLAGS -std=c++20 -DDEBUG -ggdb3 -fsanitize=address,alignment -Wall -Wextra -Weffc++ -Wc++14-compat
				 -Wmissing-declarations -Wcast-align -Wcast-qual -Wchar-subscripts -Wconversion -Wctor-dtor-privacy
				 -Wempty-body -Wfloat-equal -Wno-format-nonliteral -Wformat-security -Wformat=2 -Winline
				 -Wnon-virtual-dtor -Woverloaded-virtual -Wpacked -Wpointer-arith -Winit-self -Wredundant-decls
				 -Wshadow -Wsign-conversion -Wsign-promo -Wstrict-overflow=2 -Wsuggest-override -Wswitch-default
				 -Wundef -Wunreachable-code -Wunused -Wvariadic-macros -Wno-missing-field-initializers -Wno-narrowing
				 -Wno-old-style-cast -Wno-varargs -Wno-unused-parameter -fsized-deallocation -fstack-protector
				 -fstrict-overflow -fPIC -fPIE -fno-omit-frame-pointer -fPIE -Werror=vla -Wno-write-strings)
set (RELEASE_FLAGS -std=c++20)

find_program(IWYU_PATH NAMES include-what-you-use iwyu)

if(IWYU_PATH)
    message(STATUS "Found include-what-you-use: ${IWYU_PATH}")

    set(IWYU_COMMAND "${IWYU_PATH}"
        "-Xiwyu;--verbose=1;"
        "-Xiwyu;--comment_style=long;"
        "-isysroot;${CMAKE_OSX_SYSROOT};"
        "-Xiwyu;--mapping_file=$ENV{HOME}/iwyu/src/libcxx.imp"
    )

    set(CMAKE_CXX_INCLUDE_WHAT_YOU_USE
        "$<$<CONFIG:Debug>:${IWYU_COMMAND}>"
    )
else()
    message(STATUS "include-what-you-use not found, continuing without it")
endif()

file(GLOB SOURCES "src/*.cpp" "src/*.c")
file(GLOB HEADERS "include/*.hpp" "include/*.h")

add_library(project_objects OBJECT ${SOURCES} ${HEADERS})

target_include_directories(project_objects
    PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/include
        ${CMAKE_CURRENT_SOURCE_DIR}/src
    PUBLIC
        ${CMAKE_CURRENT_SOURCE_DIR}/include
)

target_compile_options(project_objects PRIVATE $<$<CONFIG:Debug>:${DEBUG_FLAGS}>)
target_compile_options(project_objects PRIVATE $<$<CONFIG:Release>:${RELEASE_FLAGS}>)

add_executable(${PROJECT_NAME} src/main.cpp)

target_link_libraries(${PROJECT_NAME} PRIVATE project_objects)

target_compile_options(${PROJECT_NAME} PRIVATE $<$<CONFIG:Debug>:${DEBUG_FLAGS}>)
target_compile_options(${PROJECT_NAME} PRIVATE $<$<CONFIG:Release>:${RELEASE_FLAGS}>)

target_link_options(${PROJECT_NAME} PRIVATE $<$<CONFIG:Debug>:${DEBUG_FLAGS}>)
target_link_options(${PROJECT_NAME} PRIVATE $<$<CONFIG:Release>:${RELEASE_FLAGS}>)

if(CMAKE_BUILD_TYPE STREQUAL "Debug")
    message(STATUS "Building tests for Debug configuration")

    find_package(Catch2)

    file(GLOB TEST_SOURCES "tests/*.cpp")

    if(NOT TEST_SOURCES)
        message(STATUS "No test files found in tests/ directory")
        return()
    endif()

    add_executable(tests ${TEST_SOURCES})
    if(IWYU_PATH)
        set_target_properties(tests PROPERTIES
            CXX_INCLUDE_WHAT_YOU_USE ""
        )
    endif()

    target_compile_definitions(tests PRIVATE
        TEST_DATA_DIR="${CMAKE_CURRENT_SOURCE_DIR}/tests/"
    )

    target_link_libraries(tests PRIVATE
        Catch2::Catch2WithMain
        project_objects
    )

    target_include_directories(tests PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/include
        ${CMAKE_CURRENT_SOURCE_DIR}/src
    )

    target_compile_options(tests PRIVATE ${DEBUG_FLAGS})
    target_link_options(tests PRIVATE ${DEBUG_FLAGS})

    include(Catch)
    catch_discover_tests(tests)
endif()
